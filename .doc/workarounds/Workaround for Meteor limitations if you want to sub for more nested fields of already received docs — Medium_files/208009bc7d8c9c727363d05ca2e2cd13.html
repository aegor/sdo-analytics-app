if (Meteor.isClient)
{
CC_Albums_unpublishedFields = new Meteor.Collection('CC_Albums_unpublishedFields');
CC_AlbumItems_unpublishedFields = new Meteor.Collection('CC_AlbumItems_unpublishedFields');
CC_AlbumNotes_unpublishedFields = new Meteor.Collection('CC_AlbumNotes_unpublishedFields');
}
Meteor.publish("album_edit", function(albumId)
{
check(albumId, String);
if (!(this.userId || albumId))
{
this.ready();
return;
}
var album, album_unpublishedFields,
albumItems, albumItems_unpublishedFields,
albumNote, albumNote_unpublishedFields;
album = Albums.find(
{
_id: albumId,
owner: this.userId
},
{
fields:
{
'title.unpublished': 0,
'slug.unpublished': 0
}
}
);
album_unpublishedFields = Albums.find(
{
_id: albumId,
owner: this.userId
},
{
fields:
{
owner: 1,
'title.unpublished': 1,
'slug.unpublished': 1
}
}
);
if (!album.fetch()[0])
{
this.ready();
return;
}
albumId = album.fetch()[0]._id;
albumItems = AlbumItems.find(
{
album: albumId,
owner: this.userId
},
{
fields:
{
'order.unpublished': 0
}
}
);
albumItems_unpublishedFields = AlbumItems.find(
{
album: albumId,
owner: this.userId
},
{
fields:
{
'album': 1,
'order.unpublished': 1
}
}
);
albumNote = AlbumNotes.find(
{
album: albumId,
owner: this.userId
},
{
fields:
{
'sirtrevor.unpublished': 0
}
}
);
albumNote_unpublishedFields = AlbumNotes.find(
{
album: albumId,
owner: this.userId
},
{
fields:
{
album: 1,
owner: 1,
'sirtrevor.unpublished': 1
}
}
);
Mongo.Collection._publishCursor(album_unpublishedFields, this, 'CC_Albums_unpublishedFields');
Mongo.Collection._publishCursor(albumItems_unpublishedFields, this, 'CC_AlbumItems_unpublishedFields');
Mongo.Collection._publishCursor(albumNote_unpublishedFields, this, 'CC_AlbumNotes_unpublishedFields');
return [
album,
albumItems,
albumNote
]
});
