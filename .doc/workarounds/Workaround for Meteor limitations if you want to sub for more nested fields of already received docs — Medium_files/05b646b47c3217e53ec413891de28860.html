publishAlbum = function()
{
// works for both: Publish and Publish changes
// just syncs .unpublished fields contents to .published
// and changes .published to true when needed
if (!albumEditor.albumId)
return;
// publish all unpublished items
Meteor.call('publishAlbumItems', albumEditor.albumId);
// find album note
var albumNote_unpublishedFields = CC_AlbumNotes_unpublishedFields.findOne(
{
_id: albumEditor.albumNoteId
},
{
reactive: false
}
);
// publish album note
AlbumNotes.update(
{
_id: albumEditor.albumNoteId
},
{
$set: {
'sirtrevor.published': albumNote_unpublishedFields.sirtrevor && albumNote_unpublishedFields.sirtrevor.unpublished,
published: true
}
}
);
// find album to get unpublished fields
var album_unpublishedFields = CC_Albums_unpublishedFields.findOne(
{
_id: albumEditor.albumId
},
{
reactive: false
}
);
var album = Albums.findOne(
{
_id: albumEditor.albumId
},
{
reactive: false
}
);
if (album.published)
{
// publish changes
Albums.update(
{
_id: albumEditor.albumId
},
{
$set: {
'title.published': album_unpublishedFields.title && album_unpublishedFields.title.unpublished,
'slug.published': album_unpublishedFields.slug && album_unpublishedFields.slug.unpublished
}
}
);
}
else
{
// publish unpublished album
Albums.update(
{
_id: albumEditor.albumId
},
{
$set: {
published: true,
// 1 based order
order: albumEditor.publishedAlbumsCount + 1,
'title.published': album_unpublishedFields.title && album_unpublishedFields.title.unpublished,
'slug.published': album_unpublishedFields.slug && album_unpublishedFields.slug.unpublished
}
}
);
}
};
